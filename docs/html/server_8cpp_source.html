57/**
58 * @brief Загружает базу данных клиентов из файла
59 *
60 * @details
61 * Читает файл построчно, каждая строка должна быть в формате:
62 * логин:пароль
63 *
64 * Пример файла:
65 * alice:P@ssw0rd1
66 * bob:Secret123
67 * charlie:Qwerty!@#
68 *
69 * @note Пароли хранятся в открытом виде (небезопасно!)
70 * @note Пустые строки и строки без ':' игнорируются
71 */
72void Server::load_clients() {
73 std::ifstream file(config_.client_db_file);
74 if (!file.is_open()) {
75 logger_.log_error("Cannot open client database: " + config_.client_db_file, true);
76 throw std::runtime_error("Cannot open client database");
77 }
78
79 std::string line;
80 while (std::getline(file, line)) {
81 size_t pos = line.find(':');
82 if (pos != std::string::npos) {
83 std::string login = line.substr(0, pos);
84 std::string password = line.substr(pos + 1);
85 clients_[login] = password;
86 }
87 }
88 file.close();
89
90 logger_.log("Loaded " + std::to_string(clients_.size()) + " clients");
91}

93/**
94 * @brief Настраивает серверный сокет
95 *
96 * @details
97 * Выполняет следующие действия:
98 * 1. Создает TCP сокет (AF_INET, SOCK_STREAM)
99 * 2. Устанавливает опцию SO_REUSEADDR для быстрого переиспользования порта
100 * 3. Привязывает сокет к адресу INADDR_ANY и указанному порту
101 * 4. Переводит сокет в режим прослушивания с очередью на 10 подключений
102 *
103 * @throw std::runtime_error при ошибках системных вызовов (socket, bind, listen)
104 */
105void Server::setup_socket() {
106 server_fd_ = socket(AF_INET, SOCK_STREAM, 0);
107 if (server_fd_ < 0) {
108 logger_.log_error("Socket creation failed", true);
109 throw std::runtime_error("Socket creation failed");
110 }
111
112 int opt = 1;
113 if (setsockopt(server_fd_, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt))) {
114 logger_.log_error("Setsockopt failed", true);
115 throw std::runtime_error("Setsockopt failed");
116 }
117
118 struct sockaddr_in address;
119 address.sin_family = AF_INET;
120 address.sin_addr.s_addr = INADDR_ANY;
121 address.sin_port = htons(config_.port);
122
123 if (bind(server_fd_, (struct sockaddr*)&address, sizeof(address)) < 0) {
124 logger_.log_error("Bind failed on port " + std::to_string(config_.port), true);
125 throw std::runtime_error("Bind failed");
126 }
127
128 if (listen(server_fd_, 10) < 0) {
129 logger_.log_error("Listen failed", true);
130 throw std::runtime_error("Listen failed");
131 }
132
133 logger_.log("Server socket setup complete on port " + std::to_string(config_.port));
134}

136/**
137 * @brief Принимает входящие подключения
138 *
139 * @details
140 * Бесконечный цикл, который:
141 * 1. Ожидает входящее подключение (accept)
142 * 2. Получает IP-адрес клиента для логирования
143 * 3. Создает объект Session для обработки подключения
144 * 4. Запускает обработку сессии
145 *
146 * @note Обработка блокирующая - следующее подключение ждет завершения текущего
147 * @note При ошибке accept логирует ошибку и продолжает работу
148 */
149void Server::accept_connections() {
150 struct sockaddr_in address;
151 socklen_t addrlen = sizeof(address);
152
153 logger_.log("Server started, waiting for connections...");
154
155 while (true) {
156 int client_socket = accept(server_fd_, (struct sockaddr*)&address, &addrlen);
157 if (client_socket < 0) {
158 logger_.log_error("Accept failed", false);
159 continue;
160 }
161
162 char client_ip[INET_ADDRSTRLEN];
163 inet_ntop(AF_INET, &address.sin_addr, client_ip, INET_ADDRSTRLEN);
164 logger_.log("New connection from " + std::string(client_ip));
165
166 Session session(client_socket, clients_, logger_);
167 session.handle();
168 }
169}
