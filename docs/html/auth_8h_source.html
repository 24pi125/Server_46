18/**
19 * @brief Класс аутентификации клиентов
20 *
21 * Предоставляет статические методы для работы с аутентификацией:
22 * - генерация 64-битной соли в виде 16 hex символов
23 * - вычисление MD5 хэша по схеме salt+password
24 * - верификация клиентских учетных данных
25 */
26class Authenticator {
27public:
28 /**
29 * @brief Генерирует 64-битную соль в виде 16 hex символов
30 *
31 * @return std::string 16-символьная hex строка соли
32 *
33 * @note Использует случайные числа через std::random_device
34 * @warning Соль должна быть уникальной для каждой сессии
35 */
36 static std::string generate_salt_16();
37
38 /**
39 * @brief Вычисляет MD5 хэш от конкатенации соли и пароля
40 *
41 * @param salt Соль (16 hex символов)
42 * @param password Пароль в открытом виде
43 * @return std::string 32-символьный MD5 хэш в hex формате
44 *
45 * @note Порядок: salt + password (соль добавляется перед паролем)
46 * @warning Для реальных проектов используйте более стойкие алгоритмы (SHA-256, bcrypt)
47 */
48 static std::string calculate_md5_hash(const std::string& salt,
49 const std::string& password);
50
51 /**
52 * @brief Проверяет аутентификационные данные клиента
53 *
54 * @param login Логин клиента
55 * @param received_hash Хэш, полученный от клиента
56 * @param salt Соль, использованная при аутентификации
57 * @param clients База данных клиентов (логин -> пароль)
58 * @return bool true если аутентификация успешна, иначе false
59 *
60 * @note Сравнивает полученный хэш с вычисленным MD5(salt + stored_password)
61 * @throw Ничего не выбрасывает, возвращает false при ошибках
62 */
63 static bool verify_client(const std::string& login,
64 const std::string& received_hash,
65 const std::string& salt,
66 const std::unordered_map<std::string, std::string>& clients);
67};
