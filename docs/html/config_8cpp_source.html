18 *
19 * @param argc Количество аргументов
20 * @param argv Массив аргументов
21 * @return ServerConfig Заполненная структура конфигурации
22 *
23 * @details
24 * Обрабатывает следующие опции:
25 * -h, --help -> выводит справку и завершает программу
26 * -p PORT -> устанавливает порт сервера
27 * -c FILE -> указывает файл конфигурации клиентов
28 * -d FILE -> синоним для -c
29 * -l FILE -> указывает файл логов
30 *
31 * @note При неизвестном аргументе выводит справку и завершает программу с кодом 1
32 * @note Если аргументов нет, возвращает конфигурацию по умолчанию
33 */
34ServerConfig ServerConfig::parse_args(int argc, char* argv[]) {
35 ServerConfig config;
36
37 // Если нет аргументов - возвращаем конфиг по умолчанию
38 if (argc == 1) {
39 return config; // Просто возвращаем дефолтный конфиг
40 }
41
42 for (int i = 1; i < argc; i++) {
43 if (strcmp(argv[i], "-h") == 0 || strcmp(argv[i], "--help") == 0) {
44 print_help();
45 exit(0);
46 } else if (strcmp(argv[i], "-d") == 0 && i + 1 < argc) {
47 config.client_db_file = argv[++i];
48 } else if (strcmp(argv[i], "-l") == 0 && i + 1 < argc) {
49 config.log_file = argv[++i];
50 } else if (strcmp(argv[i], "-p") == 0 && i + 1 < argc) {
51 config.port = std::stoi(argv[++i]);
52 } else if (strcmp(argv[i], "-c") == 0 && i + 1 < argc) {
53 config.client_db_file = argv[++i];
54 } else {
55 // Неизвестный аргумент
56 std::cerr << "Unknown option: " << argv[i] << "\n\n";
57 print_help();
58 exit(1);
59 }
60 }
61
62 return config;
63}
