19/**
20 * @brief Генерирует 64-битную соль в виде 16 hex символов
21 *
22 * @return std::string Соль в hex формате (16 символов, uppercase)
23 *
24 * @details
25 * Алгоритм:
26 * 1. Создает генератор случайных чисел на основе std::random_device
27 * 2. Генерирует 64-битное случайное число
28 * 3. Конвертирует в hex строку с заполнением нулями
29 *
30 * @note Использует Mersenne Twister для генерации псевдослучайных чисел
31 */
32std::string Authenticator::generate_salt_16() {
33 std::random_device rd;
34 std::mt19937_64 gen(rd());
35 std::uniform_int_distribution<uint64_t> dis;
36
37 uint64_t salt = dis(gen);
38
39 std::stringstream ss;
40 ss << std::hex << std::uppercase << std::setw(16) << std::setfill('0') << salt;
41 return ss.str();
42}

44/**
45 * @brief Вычисляет MD5 хэш от конкатенации соли и пароля
46 *
47 * @param salt Соль (должна быть 16 hex символов)
48 * @param password Пароль в открытом виде
49 * @return std::string 32-символьный MD5 хэш в hex формате (uppercase)
50 *
51 * @details
52 * Формула: hash = MD5(salt + password)
53 * Использует OpenSSL библиотеку для вычисления MD5
54 *
55 * @warning Не используйте в продакшене - MD5 уязвим к коллизиям
56 */
57std::string Authenticator::calculate_md5_hash(const std::string& salt,
58 const std::string& password) {
59 std::string data = salt + password;
60
61 unsigned char hash[MD5_DIGEST_LENGTH];
62 MD5(reinterpret_cast<const unsigned char*>(data.c_str()), data.length(), hash);
63
64 std::stringstream ss;
65 ss << std::hex << std::uppercase;
66 for (int i = 0; i < MD5_DIGEST_LENGTH; i++) {
67 ss << std::setw(2) << std::setfill('0') << static_cast<int>(hash[i]);
68 }
69
70 return ss.str();
71}

73/**
74 * @brief Проверяет аутентификационные данные клиента
75 *
76 * @param login Логин клиента
77 * @param received_hash Хэш, полученный от клиента (32 hex символа)
78 * @param salt Соль, использованная при аутентификации (16 hex символов)
79 * @param clients База данных клиентов в формате логин:пароль
80 * @return bool true если хэши совпадают, false если нет или ошибка
81 *
82 * @details
83 * Шаги проверки:
84 * 1. Поиск логина в базе данных
85 * 2. Вычисление ожидаемого хэша: MD5(salt + stored_password)
86 * 3. Сравнение полученного и вычисленного хэшей
87 *
88 * @note Возвращает false если логин не найден или хэши не совпадают
89 */
90bool Authenticator::verify_client(const std::string& login,
91 const std::string& received_hash,
92 const std::string& salt,
93 const std::unordered_map<std::string, std::string>& clients) {
94 auto it = clients.find(login);
95 if (it == clients.end()) {
96 return false;
97 }
98
99 std::string calculated_hash = calculate_md5_hash(salt, it->second);
100 return calculated_hash == received_hash;
101}
