29 // Функция расчета произведения вектора (копия из session.cpp)
30 int32_t calculate_vector_product_test(const std::vector<int32_t>& vector) {
31 if (vector.empty()) {
32 return 0;
33 }
34
35 int64_t product = 1;
36 for (int32_t val : vector) {
37 // Проверка переполнения при умножении
38 // Используем static_cast<int64_t> для безопасного преобразования
39 int64_t val64 = static_cast<int64_t>(val);
40
41 if (val64 != 0 && llabs(product) > INT64_MAX / llabs(val64)) {
42 // Переполнение
43 if ((product > 0 && val64 > 0) || (product < 0 && val64 < 0)) {
44 return INT32_MAX;
45 } else {
46 return INT32_MIN;
47 }
48 }
49 product *= val64;
50 }
51
52 // Проверка выхода за пределы int32
53 if (product > INT32_MAX) {
54 return INT32_MAX;
55 } else if (product < INT32_MIN) {
56 return INT32_MIN;
57 }
58
59 return static_cast<int32_t>(product);
60 }
